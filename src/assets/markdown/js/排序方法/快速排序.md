> 快速排序

```
 //  快速排序
var arr = [19, 9, 19, 10, 2, 3, 5, 7, 1, 1, 1, 4, 5, 6, 9, 7];
function foo(arr) {
    if (arr.length <= 1) {
    return arr;
    }
    var index = Math.floor(arr.length / 2);
    var num = arr[index];
    var arrL = [];
    var arrC = [];
    var arrR = [];
    for (var i = 0; i < arr.length; i++) {
    if (arr[i] < num) {
        arrL.push(arr[i]);
    } else if (arr[i] > num) {
        arrR.push(arr[i]);
    } else {
        arrC.push(arr[i]);
    }
    }
    return [...foo(arrL), ...arrC, ...foo(arrR)];
}
console.log(foo(arr));
```
